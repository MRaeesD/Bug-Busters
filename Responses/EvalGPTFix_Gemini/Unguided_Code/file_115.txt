import java.util.*;
    class Main {
        public static void main(String args[]){
            Scanner sc = new Scanner(System.in);
            long A = sc.nextLong();
            long B = sc.nextLong();
            int cnt = 0;
            while(A!=B){
                if(A>B){
                    long div = A/B;
                    A = A-B*div;
                    if(A==0){
                        cnt += div-1;
                        break;
                    }
                    cnt += div;
                    continue;
                }else if(A<B){
                    long div = B/A;
                    B = B-A*div;
                    if(B==0){
                        cnt += div-1;
                        break;
                    }
                    cnt += div;
                    continue;
                }
            }
            System.out.println(cnt);
        }
    }
