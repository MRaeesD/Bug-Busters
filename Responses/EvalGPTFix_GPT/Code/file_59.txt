import java.util.Scanner;
            public class Main {
                public static void main(String[] args) {
                    Scanner get = new Scanner(System.in);
                    int N = get.nextInt();
                    String S = get.next();
                    get.close();
                    
                    int a = 0, b = 0;
                    for (int i = 0; i < N; i++) {
                        String A = S.substring(i, i + 1);  
                        
                        if (A.equals("x")) {
                            b = 1;
                        } else if (A.equals("o")) {
                            a = 1;
                        }
                    }
                    
                    String result;
                    if (a == 1 && b == 0) {
                        result = "Yes";
                    } else {
                        result = "No";
                    }

                    System.out.println(result);
                }
            }
