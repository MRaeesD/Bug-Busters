import java.util.*;

class Main {
    public static void main(String args[]) {
        Scanner sc = new Scanner(System.in);
        long A = sc.nextLong();
        long B = sc.nextLong();
        int cnt = 0;
        while (A != B) {
            if (A > B) {
                long div = (A - 1) / B; // compute maximum useful full subtractions
                A = A - B * div;
                cnt += div;
            } else {
                long div = (B - 1) / A; // compute maximum useful full subtractions
                B = B - A * div;
                cnt += div;
            }
        }
        System.out.println(cnt);
    }
}
