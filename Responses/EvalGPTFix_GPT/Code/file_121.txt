import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        long a = sc.nextLong(); 
        long b = sc.nextLong();
        long cnt = 0;
        while (a != b) {
            if (a > b) {
                if (b == 0) {
                    System.out.println("Error: Division by zero");
                    return;
                }
                cnt += (a - 1) / b;
                a -= (a - 1) / b * b;
            } else {
                if (a == 0) {
                    System.out.println("Error: Division by zero");
                    return;
                }
                cnt += (b - 1) / a;
                b -= (b - 1) / a * a;
            }
        }
        System.out.println(cnt);
    }
}
