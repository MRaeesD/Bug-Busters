import java.util.Scanner;

            public class Main {
                public static void main(String[] args) {
                    Scanner sc = new Scanner(System.in);
                    long a = sc.nextLong(); 
                    long b = sc.nextLong();
                    long cnt = 0;
                    while (a != b && a > 0 && b > 0) {
                        cnt++;
                        if (a > b) {
                            cnt += (a - 1) / b;
                            a -= (a - 1) / b * b;
                        } else {
                            cnt += (b - 1) / a;
                            b -= (b - 1) / a * a;
                        }
                    }
                    if (a == 0 || b == 0) {
                        System.out.println("Operation not possible due to zero value");
                    } else {
                        System.out.println(cnt);
                    }
                }
            }
