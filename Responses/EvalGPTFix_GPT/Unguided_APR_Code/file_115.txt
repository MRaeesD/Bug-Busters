import java.util.*;
            class Main {
                public static void main(String args[]) {
                    Scanner sc = new Scanner(System.in);
                    long A = sc.nextLong();
                    long B = sc.nextLong();
                    int cnt = 0;
                    while (A != B) {
                        if (A > B) {
                            long div = A / B;
                            if (A % B == 0) {
                                cnt += div;
                                break;
                            }
                            A = A % B;
                            cnt += div;
                        } else {
                            long div = B / A;
                            if (B % A == 0) {
                                cnt += div;
                                break;
                            }
                            B = B % A;
                            cnt += div;
                        }
                    }
                    System.out.println(cnt);
                }
            }
