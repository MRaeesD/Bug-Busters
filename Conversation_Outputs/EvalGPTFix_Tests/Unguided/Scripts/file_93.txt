import java.util.*;

public class Main {
  public static void main(String args[]) {
    Scanner sc = new Scanner(System.in);
    String s = sc.next();
    String ans = "YES"; 

    // Input validation
    if (s.length() != 8) {
      System.out.println("NO"); // Output NO if the input length is not 8
      return;
    }

    int firstBIndex = -1;
    int secondBIndex = -1;
    int firstRIndex = -1;
    int secondRIndex = -1;
    int kIndex = -1;

    for (int i = 0; i < 8; i++) {
      char ch = s.charAt(i);
      if (ch == 'B') {
        if (firstBIndex == -1) {
          firstBIndex = i;
        } else {
          secondBIndex = i;
        }
      } else if (ch == 'R') {
        if (firstRIndex == -1) {
          firstRIndex = i;
        } else {
          secondRIndex = i;
        }
      } else if (ch == 'K') {
        kIndex = i;
      }
    }

    // Check 'B' placement logic
    if (Math.abs(secondBIndex - firstBIndex) % 2 == 0) {
      ans = "NO"; // 'B' characters must be on alternating indices
    }

    // Check 'K' placement logic
    if (!(firstRIndex < kIndex && kIndex < secondRIndex)) {
      ans = "NO"; // 'K' must be between two 'R's
    }

    System.out.println(ans);
  }
}
