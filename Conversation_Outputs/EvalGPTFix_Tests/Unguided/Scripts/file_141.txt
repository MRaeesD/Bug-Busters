import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        boolean result = false; // Start with false, set to true only if a valid transformation is found
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[][] a = new int[n][n];
        int[][] b = new int[n][n];
        for(int i = 0; i < n;i++) {
            for(int j = 0; j < n; j++) {
                a[i][j] = sc.nextInt();
            }
        }
        for(int i = 0; i < n;i++) {
            for(int j = 0; j < n; j++) {
                b[i][j] = sc.nextInt();
            }
        }

        // Check for 0-degree rotation (no rotation)
        result = true;
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < n; j++) {
                if(a[i][j] != b[i][j]) {
                    result = false;
                    break;
                }
            }
            if(!result) break;
        }

        // Check for 90-degree clockwise rotation
        if(!result) {
            result = true;
            for(int i = 0; i < n; i++) {
                for(int j = 0; j < n; j++) {
                    if(a[n-1-j][i] != b[i][j]) {
                        result = false;
                        break;
                    }
                }
                if(!result) break;
            }
        }

        // Check for 180-degree rotation
        if(!result) {
            result = true;
            for(int i = 0; i < n; i++) {
                for(int j = 0; j < n; j++) {
                    if(a[n-1-i][n-1-j] != b[i][j]) {
                        result = false;
                        break;
                    }
                }
                if(!result) break;
            }
        }

        // Check for 270-degree clockwise rotation
        if(!result) {
            result = true;
            for(int i = 0; i < n; i++) {
                for(int j = 0; j < n; j++) {
                    if(a[j][n-1-i] != b[i][j]) {
                        result = false;
                        break;
                    }
                }
                if(!result) break;
            }
        }

        if(result) {
            System.out.println("Yes");
        } else {
            System.out.println("No");
        }
    }
}
