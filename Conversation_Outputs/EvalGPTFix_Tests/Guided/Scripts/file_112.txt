import java.util.*;

public class Main {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        Long A = sc.nextLong();
        Long B = sc.nextLong();
        int ans = 0;

        while (A != 0 && B != 0) {
            if (A > B) {
                ans += A / B;
                A = A % B;
            } else {
                ans += B / A;
                B = B % A;
            }
        }
        
        System.out.println(ans);
    }
}
