import java.util.*;
class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        int D = sc.nextInt();
        int ans = -1; // Initialize to -1
        int pre = 0;
        for(int i=1; i<=N; i++){
            int T = sc.nextInt();
            if(i>1 && (T-pre)<= D && (ans == -1 || T < ans)){ // Find the smallest T
                ans = T; 
            } 
            pre = T; 
        }
        
        if(ans == -1) {
            System.out.println("No valid pair found"); 
        } else {
            System.out.println(ans);
        }
    }
}
