import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[][] a = new int[n][n];
        int[][] b = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                a[i][j] = sc.nextInt();
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                b[i][j] = sc.nextInt();
            }
        }
        boolean result = false;
        for (int t = 0; t < 4; t++) {
            if (check(a, b, n)) {
                result = true;
                break;
            }
            // Rotate matrix a by 90 degrees
            a = rotate(a, n); 
        }
        if (result) {
            System.out.println("Yes");
        } else {
            System.out.println("No");
        }
    }

    // Function to check if matrices a and b are equal
    static boolean check(int[][] a, int[][] b, int n) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (a[i][j] != b[i][j]) {
                    return false;
                }
            }
        }
        return true;
    }

    // Function to rotate matrix a by 90 degrees
    static int[][] rotate(int[][] a, int n) {
        int[][] temp = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                temp[j][n - 1 - i] = a[i][j];
            }
        }
        return temp;
    }
}
