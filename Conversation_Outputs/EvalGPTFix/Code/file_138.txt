public static String solve(int n, int[][] a, int[][] b){
    for (int k = 0; k < 4; k++) { 
        boolean match = true;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (a[i][j] != b[i][j]) {
                    match = false;
                    break;
                }
            }
            if (!match) break;
        }
        if (match) {
            return "Yes";
        }
        
        // Rotate matrix a by 90 degrees clockwise
        int[][] temp = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                temp[i][j] = a[n-j-1][i];
            }
        }
        a = temp;
    }
    return "No";
}
