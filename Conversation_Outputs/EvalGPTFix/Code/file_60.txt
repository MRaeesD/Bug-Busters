import java.util.*;
public class Main {
  public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int N = sc.nextInt();
    String S = sc.next();
    
    boolean ans = false;
    
    for(int i = 0; i < N; i++) { 
      if(S.charAt(i) == 'o') { 
        ans = true;
        break; 
      }
    }
    
    if(ans) {
      System.out.println("Yes");
    } else {
      System.out.println("No");
    }
  }
}
