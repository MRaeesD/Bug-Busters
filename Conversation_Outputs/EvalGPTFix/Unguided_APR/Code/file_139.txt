import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt(); 
        int[][] A = new int[N][N]; 
        for (int i = 0; i < N; i++) { 
            for (int j = 0; j < N; j++) { 
                A[i][j] = sc.nextInt();
            }
        }

        int[][] B = new int[N][N]; 
        for (int i = 0; i < N; i++) { 
            for (int j = 0; j < N; j++) { 
                B[i][j] = sc.nextInt();
            }
        }

        boolean bool = false;
        for (int n = 0; n < N; n++) {
            boolean tempbool = true;
            loop: for (int i = 0; i < N; i++) { 
                for (int j = 0; j < N; j++) { 
                    if (A[i][j] == 1) {
                        if (B[i][j] != 1) {
                            tempbool = false;
                            break loop;
                        }
                    }
                }
            }
            bool = tempbool;
            if (bool) {
                break;
            }
            int[][] temp = new int[N][N]; 
            for (int i = 0; i < N; i++) { 
                for (int j = 0; j < N; j++) { 
                    temp[j][N - 1 - i] = A[i][j]; 
                }
            }
            A = temp; 
        }
        if (bool) {
            System.out.println("Yes");
        } else {
            System.out.println("No");
        }
    }
}
