import java.util.*;
public class Main{
  public static void main(String[] args){
    Scanner sc =new Scanner(System.in);
    int N = sc.nextInt();
    String S = sc.next();
    int m = 0;
    for(int i = 0; i < N; i++){
      if(S.charAt(i) == 'x'){
        System.out.println("No");
        System.exit(0);
      } else if(S.charAt(i) == 'o'){
        m = m + 1;
      }
    }
    if(m > 0){
      System.out.println("Yes");
    }
  }
}
