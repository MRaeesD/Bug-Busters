import java.util.*;
class Main {
    public static void main(String args[]){
        Scanner sc = new Scanner(System.in);
        long A = sc.nextLong();
        long B = sc.nextLong();
        int cnt = 0; 
        while(A != B){
            if (A == 0 || B == 0) break; // Handle the case where one number becomes zero
            if (A > B) {
                cnt += A / B;
                A %= B; 
            } else {
                cnt += B / A;
                B %= A;
            }
        }
        System.out.println(cnt);
    }
}
