import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        boolean result = false; // Start with false
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[][] a = new int[n][n];
        int[][] b = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                a[i][j] = sc.nextInt();
            }
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                b[i][j] = sc.nextInt();
            }
        }

        // Check all orientations
        result = checkMatrices(a, b, n, 0) || // Original
                 checkMatrices(a, b, n, 1) || // 90-degree rotation
                 checkMatrices(a, b, n, 2) || // 180-degree rotation
                 checkMatrices(a, b, n, 3);  // 270-degree rotation 

        if (result) {
            System.out.println("Yes");
        } else {
            System.out.println("No");
        }
    }

    private static boolean checkMatrices(int[][] a, int[][] b, int n, int rotation) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int x, y;
                switch (rotation) {
                    case 1:  // 90 degrees
                        x = j;
                        y = n - 1 - i;
                        break;
                    case 2: // 180 degrees
                        x = n - 1 - i;
                        y = n - 1 - j;
                        break;
                    case 3: // 270 degrees
                        x = n - 1 - j;
                        y = i;
                        break;
                    default: // 0 degrees
                        x = i;
                        y = j;
                        break;
                }
                if (a[x][y] != b[i][j]) {
                    return false;
                }
            }
        }
        return true;
    }
}
