import java.util.Scanner;

    public class Main {

    	public static void main(String[] args) {
    		Scanner sc = new Scanner(System.in);
    		long A = sc.nextLong();
    		long B = sc.nextLong();
    		long counter = 0;

    		while (A != B) {
    			if (A > B) {
    				long wari = A / B;
    				counter += wari;
    				A -= B * wari;
    			} else if (A < B) {
    				long wari = B / A;
    				counter += wari;
    				B -= A * wari;
    			}
    		}

    		System.out.println(counter);
    	}
    }
