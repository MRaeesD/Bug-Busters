import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		long A = sc.nextLong();
		long B = sc.nextLong();
		long counter = 0;

		while (A != B) {
			if (A > B) {
				counter += A / B; // This line ensures A is reduced at least to be smaller than B
				A %= B;
			} else if (A < B) {
				counter += B / A; // This line ensures B is reduced at least to be smaller than A
				B %= A;
			}
		}

		System.out.println(counter);
	}

}
