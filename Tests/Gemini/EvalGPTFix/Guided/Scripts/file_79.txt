import java.util.*;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        String s = sc.next();
        boolean flag = false;
        for(int i=0;i<n;++i){
            if(s.charAt(i) == 'o') {
                flag = true;
            }
        }
        if(flag) System.out.println("YES"); // bug fix: Moved the check for 'o' outside the loop
        else System.out.println("NO");
    }
}
