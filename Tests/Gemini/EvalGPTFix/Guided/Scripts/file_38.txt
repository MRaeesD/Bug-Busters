import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int N=sc.nextInt();
		int D=sc.nextInt();
		int[] sample=new int[N];
		for(int i=0;i<N;i++){
		    sample[i]=sc.nextInt();
		}
		int flag=0; //change made here
		for(int i=0;i<N-1;i++){ 
		    if(sample[i+1]-sample[i]<=D){
		        System.out.println(sample[i+1]);
		        flag=1; //change made here
		        break;
		    }
		}
		if(flag==0){ //change made here
		    System.out.println(-1);
		}
	}
}
