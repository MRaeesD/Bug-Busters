import java.util.*;
public class Main{

public static void main(String[] args) {
    

    Scanner sc=new Scanner(System.in);

    int N=sc.nextInt();

    String s=sc.next();

    int len=s.length();

    String ans="NO";

    for(int i=0;i<len-1;i++){ //iterate only till the second last character
        if(s.charAt(i)=='o' && s.charAt(i+1)=='x'){ //Check for consecutive 'o' and 'x'
            ans="YES";
            break;
        }
    }

    System.out.println(ans);
}

}
