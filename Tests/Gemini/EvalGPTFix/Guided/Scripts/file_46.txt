import java.util.Scanner;

public class Main { 
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        int N = input.nextInt();
        int D = input.nextInt();
        int last = 0;
        int ans = -1; // Change: Initialize ans to -1
        for(int i = 0;i < N;i++){
            int num = input.nextInt();
            if(last != 0){
                int dis = num - last;
                if(dis <= D){
                    ans = num; // Change: Update ans instead of printing and returning
                    break; // Exit the loop after finding a valid pair
                }
            }
            last = num;
        }
        System.out.println(ans); // Change: Print ans after iterating over all numbers
    }
}
