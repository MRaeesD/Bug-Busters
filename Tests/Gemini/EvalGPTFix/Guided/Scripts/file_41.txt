import java.util.*;
class Main{
  public static void main(String args[]){ //bug fixed: Changed args to args[]
    Scanner sc=new Scanner(System.in);
    int n=sc.nextInt();
    int d=sc.nextInt();
    int a[]=new int[n];
    for(int i=0;i<n;i++)
    {
      a[i]=sc.nextInt();
    }
    int c=-1;
    for(int i=0;i<n-1;i++) // bug fixed: Changed the starting index from 1 to 0
    {
      if(a[i]-a[i-1]<=d)
      {
        c=i;break;
      }
    }
    System.out.println(c);
  }
}
