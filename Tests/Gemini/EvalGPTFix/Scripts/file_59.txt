import java.util.Scanner;
    public class Main{
        public static void main(String[] ags){
            Scanner get = new Scanner(System.in);
            int N = get.nextInt();
            String S = get.next();
            get.close();
            System.out.println(S);
            
            String A = " ";
            boolean hasX = false;
            boolean hasO = false;
            
            for(int i = 0; i < N; i++){
                A = String.valueOf(S.charAt(i)); 
                if(A.equals("x")){
                    hasX = true;
                } else if(A.equals("o")) {
                    hasO = true;
                }
            }
            if(hasX && hasO) A = "Yes";
            else A = "No";
            System.out.println(A);
        }
    }
