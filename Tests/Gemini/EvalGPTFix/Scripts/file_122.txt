import java.util.Scanner;

    public class Main {
        public static void main(String[] args) {
            Scanner sc = new Scanner(System.in);
            long a = sc.nextInt();
            long b = sc.nextInt();
            long cnt = 0;
            while (a != b) {
                cnt++;
                if (a > b) {
                    a = a % b; 
                } else {
                    b = b % a; 
                }
            }
            System.out.println(cnt);
        }
    }
